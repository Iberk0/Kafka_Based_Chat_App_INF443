# Sistem güncellemeleri
sudo apt update && sudo apt upgrade -y

# Python ve pip kurulumu
sudo apt install -y python3 python3-pip python3-venv

# Flask ve FastAPI için gerekli kütüphaneler
pip3 install flask fastapi uvicorn

# Docker kurulumu
sudo apt install -y docker.io docker-compose
sudo systemctl start docker
sudo systemctl enable docker
sudo usermod -aG docker $USER  # Docker komutlarını sudo olmadan çalıştırabilmek için

# Kafka için gerekli Docker imajını çekmek (Confluent Kafka örneği)
docker pull confluentinc/cp-kafka:latest

# Projede Kafka kullanımı için gerekli Python kütüphanesi
pip3 install confluent-kafka

# Geliştirme için Git kurulumu
sudo apt install -y git

# Docker Compose yüklü değilse
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# Proje dosyalarının geliştirilmesi için bir IDE veya metin editörü (VS Code önerilir)
sudo apt install -y code  # Eğer yüklü değilse
